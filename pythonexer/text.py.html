def run_files():
    while True:
        try:
            choice = int(input("请输入想运行的文件编号（注意：只能输入1、2、3）："))
            if choice == 1:
                os.system("exer1.py")
            elif choice == 2:
                os.system("exer2.py")
            elif choice == 3:
                os.system("exer3.py")
            else:
                print("只能输入1，2，3，请重新输入哦！！")
        except ValueError:
            print("注意：只能输入数字！")

if __name__ == "__main__":
    run_files()
